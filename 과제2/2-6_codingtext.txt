// 2-6. 컨트롤 제트
#include <string>
#include <vector>
#include <sstream>

using namespace std;

int solution(string sInput) {
    int answer = 0;
    stringstream ss(sInput);
    string sWord;
    vector<string> vStr;
    
    while(ss >> sWord) {
        vStr.push_back(sWord);
    }
    
    string str;
    for(int idx = vStr.size()-1; idx >=0; idx--)
    {
        str = vStr[idx];
        if(str == "Z") {
            idx--;
        }
        else {
            answer += stoi(str);
        }
    }
    
    return answer;
}
