// 2-5. 이차원 배열 대각선 순회하기
#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> vvIntBoard, int nK)
{
    int answer = 0;
    
    for(size_t j = 0; j < vvIntBoard.size(); j++) {
        for(size_t i = 0; i < vvIntBoard[j].size(); i++) 
        {
            if(i + j <= nK) {
                answer += vvIntBoard[j][i];
            }
            else {
                break;
            }
        }
    }
    
    return answer;
}
