// 2-4. 더 크게 합치기
#include <string>
#include <vector>

using namespace std;

int solution(int nA, int nB) {
    int answer = 0;
    string strA = to_string(nA);
    string strB = to_string(nB);
    
    string strAB = strA + strB;
    string strBA = strB + strA;
    
    answer = stoi(strAB) >= stoi(strBA) ? stoi(strAB) : stoi(strBA);
    
    return answer;
}
